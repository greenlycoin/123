using System;
using System.Data.SqlClient;
using System.Windows.Forms;
using static System.Windows.Forms.VisualStyles.VisualStyleElement;
namespace Test
{
    public partial class Form1 : Form
    {
        string connectionString = @"Data Source=(localdb)\MSSQLLocalDB;Initial Catalog=zavtra;Integrated Security=True;TrustServerCertificate=True";
        public Form1()
        {
            InitializeComponent();
            button1.Click += button1_Click;
        }

        private void button1_Click(object sender, EventArgs e)
        {

            if (string.IsNullOrWhiteSpace(textBox1.Text) || string.IsNullOrWhiteSpace(textBox2.Text))
            {
                MessageBox.Show("Введите логин и пароль", "Ошибка",
                    MessageBoxButtons.OK, MessageBoxIcon.Warning);
                return;
            }
            try
            {
                using (SqlConnection con = new SqlConnection(connectionString))
                {
                    con.Open();
                    string sql = "SELECT RoleId, Surname, Name, Pantomic FROM [User] " +
"WHERE Login=@log AND Password=@pass";
                    SqlCommand cmd = new SqlCommand(sql, con);
                    cmd.Parameters.AddWithValue("@log", textBox1.Text);
                    cmd.Parameters.AddWithValue("@pass", textBox2.Text);
                    SqlDataReader reader = cmd.ExecuteReader();
                    if (reader.Read())
                    {
                        int roleId = Convert.ToInt32(reader["RoleId"]);
                        string fio =
                        reader["Surname"].ToString() + " " +
                        reader["Name"].ToString() + " " +
                        reader["Pantomic"].ToString();
                        MessageBox.Show("Вход выполнен успешно!");
                        Form2 f = new Form2(roleId, fio);
                        f.Show();
                        this.Hide();
                    }
                    else
                    {
                        MessageBox.Show("Неверный логин или пароль");
                    }
                }
            }
        }
    }
}


using System;
using System.Data;
using System.Data.SqlClient;

using System.Windows.Forms;
namespace Test
{
    public partial class Form2 : Form
    {
        int roleId;
        string fio;
        string connectionString = @"Data Source=(localdb)\MSSQLLocalDB;Initial Catalog=zavtra;Integrated Security=True;TrustServerCertificate=True";
        public Form2(int roleId, string fio)
        {
            InitializeComponent();
            this.roleId = roleId;
            this.fio = fio;
            label1.Text = fio;
            ApplyPermissions();
            LoadProducts();
        }
        private void ApplyPermissions()
        {
            // Допустим: 1 = Админ, 2 = Менеджер, 3 = Покупатель
            switch (roleId)
            {
                case 1: // Админ
                        // Все кнопки видны
                    break;
                case 2: // Менеджер
                        // Скрыть удаление товаров
                        // Например: btnDelete.Visible = false;
                    break;
                case 3: // Покупатель
                        // Скрыть всё кроме просмотра
                        // Например: panelAdminTools.Visible = false;
                    break;
                default:
                    MessageBox.Show("Неизвестная роль. Доступ ограничен.");
                    break;
            }
        }
        private void LoadProducts()
        {
            flowLayoutPanel1.Controls.Clear();
            try
            {
                using (SqlConnection con = new SqlConnection(connectionString))
                {
                    con.Open();
                    // ВАЖНО: УБРАЛИ UnitId, чтобы не было ошибки
                    string sql =
                    "SELECT p.Id, p.Name, p.Description, " +
                   "pr.Name AS Producer, pv.Name AS Provider, " +
                   "p.Price, p.Discount, p.AmountStock, p.Photo " +
                   "FROM dbo.Product p " +
                   "JOIN dbo.Producer pr ON p.ProducedId = pr.Id " +
                   "JOIN dbo.Provider pv ON p.ProvidedId = pv.Id";
                    SqlDataAdapter da = new SqlDataAdapter(sql, con);
                    DataTable dt = new DataTable();
                    da.Fill(dt);
                    foreach (DataRow row in dt.Rows)
                    {
                        Producties card = new Producties();
                        card.FillData(
                        row["Name"].ToString(),
                       row["Description"].ToString(),
                       row["Producer"].ToString(),
                       row["Provider"].ToString(),
                       Convert.ToDecimal(row["Price"]),
                        Convert.ToInt32(row["Discount"]),
                        "шт.", // временно вместо UnitId
                        Convert.ToInt32(row["AmountStock"]),
                        row["Photo"].ToString()
                        );
                        flowLayoutPanel1.Controls.Add(card);
                    }
                }
            }
            catch (Exception ex)
            {
                MessageBox.Show("Ошибка загрузки товаров:\n" + ex.Message);
            }
        }
    }
}
using System;
using System.Drawing;
using System.IO;
using System.Windows.Forms;
namespace Test
{
    public partial class Producties : UserControl
    {
        public Producties()
        {
            InitializeComponent();
        }
        public void FillData(
        string title,
        string desc,
        string producer,
        string provider,
        decimal price,
        int discount,
        string unit,
        int stock,
        string photo
        )
        {
            // Текст
            lblTitle.Text = title;
            lblDesc.Text = desc;
            lblProducer.Text = "Производитель: " + producer;
            lblProvider.Text = "Поставщик: " + provider;
            // Цена
            lblOldPrice.Text = price.ToString("0.00");
            decimal newPrice = price - (price * discount / 100m);
            lblPrice.Text = newPrice.ToString("0.00");
            lblUnit.Text = "Ед.: " + unit;
            lblStock.Text = "Остаток: " + stock;
            lblDiscount.Text = "-" + discount + "%";
            // Фото: ожидается папка Images рядом с exe
            // Application.StartupPath = папка, где лежит exe
            string imgPath = Path.Combine(Application.StartupPath, "Images", photo);
            if (File.Exists(imgPath))
                pbPhoto.Image = Image.FromFile(imgPath);
            else
                pbPhoto.Image = null;
            // Цвет карточки (как в гайде логикой)
            if (stock == 0)
                BackColor = Color.LightBlue;
            else if (discount > 15)
                BackColor = Color.LightGreen;
            else
                BackColor = Color.White;
        }
    }
}
